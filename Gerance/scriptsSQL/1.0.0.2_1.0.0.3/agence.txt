DROP TABLE if exists agence.workflow;
CREATE TABLE agence.workflow
(
	id character varying(32) not null default public.get_uuid(),
	reference int not null,
	date_reference date,
	nom  character varying, 
	revision integer,
	statut integer,
	audit_created timestamp without time zone default now(),
	audit_created_by character varying,
	audit_updated timestamp without time zone,
	audit_updated_by character varying,
  
	CONSTRAINT workflow_pkey PRIMARY KEY (id)
)
WITH (
  OIDS=FALSE
);
ALTER TABLE agence.workflow
  OWNER TO postgres;
GRANT ALL ON TABLE agence.workflow TO postgres;
GRANT SELECT, UPDATE, INSERT, TRUNCATE, DELETE ON TABLE agence.workflow TO public;
CREATE UNIQUE INDEX workflow_reference_date_reference_idx
  ON agence.workflow
  USING btree
  (reference , date_reference);
