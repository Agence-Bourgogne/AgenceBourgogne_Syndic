
DROP TABLE if exists users;
CREATE TABLE users
(
	id character varying(32) not null default public.get_uuid(),
	reference character varying not null,
	password character varying,
	nom character varying not null,
	prenom character varying(50),  
	roles_id character varying not null,
	resources_id character varying,
	statut integer,
	audit_created timestamp without time zone default now(),
	audit_created_by character varying,
	audit_updated timestamp without time zone,
	audit_updated_by character varying,
  
	CONSTRAINT users_pkey PRIMARY KEY (id)
)
WITH (
  OIDS=FALSE
);
GRANT ALL ON TABLE users TO postgres;
GRANT SELECT, UPDATE, INSERT, TRUNCATE, DELETE ON TABLE users TO public;

CREATE UNIQUE INDEX user_reference_idx
  ON users
  USING btree
  (reference COLLATE pg_catalog."default");

DROP TABLE if exists roles;
CREATE TABLE roles
(
	id character varying(32) not null default public.get_uuid(),
	reference character varying not null,
	nom character varying,

	audit_created timestamp without time zone default now(),
	audit_created_by character varying,
	audit_updated timestamp without time zone,
	audit_updated_by character varying,

	CONSTRAINT roles_pkey PRIMARY KEY (id)
)
WITH (
  OIDS=FALSE
);

GRANT ALL ON TABLE roles TO postgres;
GRANT SELECT, UPDATE, INSERT, TRUNCATE, DELETE ON TABLE roles TO public;

CREATE UNIQUE INDEX roles_reference_idx
  ON roles
  USING btree
  (reference COLLATE pg_catalog."default");
