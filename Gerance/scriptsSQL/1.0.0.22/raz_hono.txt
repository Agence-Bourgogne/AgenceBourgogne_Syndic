update agence.biens set honoraires_locataire = 0
where id in
(
	select bien_id
	from agence.quittances
	where date_quittance >= '2015-08-01' 
	and (montant_loyer+ montant_augmentation+ valeur_taxe+ montant_charge+ frais_bail + montant_divers1 + montant_divers2+ montant_divers3 + montant_divers4+ montant_divers5 + honoraire_locataire) != montant_quittance
);

update agence.quittances set honoraire_locataire=0
where date_quittance >= '2015-08-01' 
and (montant_loyer+ montant_augmentation+ valeur_taxe+ montant_charge+ frais_bail + montant_divers1 + montant_divers2+ montant_divers3 + montant_divers4+ montant_divers5 + honoraire_locataire) != montant_quittance
;
