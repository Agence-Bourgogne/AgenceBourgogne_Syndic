drop table if exists parametres;

create table parametres (
  id character varying(32) not null default public.get_uuid(),
  groupe character varying(20),
  code character varying(20),
  nom character varying(80),
  param_1 character varying(20),
  param_2 character varying(20),
  param_3 character varying(20),
  iparam_1 integer,
  iparam_2 integer,
  iparam_3 integer,
CONSTRAINT parametres_pkey PRIMARY KEY (id)
)
WITH (
  OIDS=FALSE
);
ALTER TABLE parametres
  OWNER TO postgres;
GRANT SELECT, UPDATE, INSERT, DELETE ON TABLE parametres TO public;

-- Function: create_immeuble(character varying, character varying)

-- DROP FUNCTION create_immeuble(character varying, character varying);

CREATE OR REPLACE FUNCTION create_immeuble(code_agence character varying, ref_immeuble character varying)
  RETURNS boolean AS
$BODY$
DECLARE 
	rcode boolean := 1;
	ref_schema character varying;
	table_dst character varying;
	table_src character varying;
	curr_table character varying;
	base_agence character varying;
	cmd character varying;
	
BEGIN
	ref_schema:=lower(code_agence || '_' || ref_immeuble);
	raise notice 'create schema(%)', ref_schema;
		execute 'create schema "' || ref_schema ||'"';
		execute 'grant all on schema "' || ref_schema || '" TO public';
		base_agence := 'agence_base';
		select_tables:= 'select tablename from pg_tables where schemaname = ''' || base_agence ||'''';
		for schema_cursor in execute select_tables
		loop
			curr_table := schema_cursor.tablename;
			table_src := base_agence || '.' || curr_table;
			table_dst := code_agence || '_' || ref_immeuble || '.' || curr_table;
			cmd := 'create table @table_dst (like  @table_src including all)';
			cmd := replace (cmd, '@table_dst', table_dst);
			cmd := replace (cmd, '@table_src', table_src);
			raise notice  'execute %', cmd;
			execute cmd;
			cmd := 'GRANT SELECT, UPDATE, INSERT, DELETE ON TABLE '||table_dst|| ' TO public';

		end loop;
	return rcode;
END;
$BODY$
  LANGUAGE plpgsql VOLATILE
  COST 100;
ALTER FUNCTION create_immeuble(character varying, character varying)
  OWNER TO postgres;

create or replace function uuid_generate_uuid()
returns uuid as
$BODY$ declare 
begin
	return uuid_in(md5(random()::text || now()::text)::cstring);
end; $BODY$
language plpgsql volatile strict cost 1;
ALTER FUNCTION uuid_generate_uuid()
  OWNER TO postgres;
  
  CREATE OR REPLACE FUNCTION get_uuid()
  RETURNS character varying AS
$BODY$ declare
var varchar:= uuid_generate_uuid();
begin
while var = uuid_generate_uuid()::varchar loop
end loop;
return replace(upper(uuid_generate_v4()::varchar),'-','');
end; $BODY$
  LANGUAGE plpgsql VOLATILE
  COST 100;

  OWNER TO postgres;
