DROP SCHEMA agence;

CREATE SCHEMA agence
AUTHORIZATION postgres;

GRANT ALL ON SCHEMA agence TO postgres;
GRANT ALL ON SCHEMA agence TO public;
COMMENT ON SCHEMA agence
  IS 'schema standard agence';


DROP TABLE if exists agence.coproprietaire;
CREATE TABLE agence.coproprietaire
(
	id character varying(32) not null default public.get_uuid(),
	reference character varying not null,
	nom character varying not null,
	vente boolean,
	prenom character varying(50),  
	codenvoi integer,
	adresse character varying(255),
	codepostal character varying(5),
	ville character varying(50),
	telephone character varying(16),
	email character varying,
	pays character varying,
	nomcomp character varying(50),
	codenvcomp integer,
	adressecomp character varying(255),
	villecomp character varying(50),
	codecomp character varying(5),
	telcomp character varying(15),
	declaration boolean,
	drapeau varchar(5),
	note character varying(255),
	dateappel date,
	daterel1 date,
	daterel2 date,
	daterel3 date,
	statut integer,
	commerce boolean,
	audit_created timestamp without time zone default now(),
	audit_created_by character varying(35),
	audit_updated timestamp without time zone,
	audit_updated_by character varying(35),
  
	CONSTRAINT coproprietaire_pkey PRIMARY KEY (id)
)
WITH (
  OIDS=FALSE
);
ALTER TABLE agence.coproprietaire
  OWNER TO postgres;
GRANT SELECT, UPDATE, INSERT, DELETE ON TABLE agence.coproprietaire TO public;
CREATE UNIQUE INDEX coproprietaire_reference_idx
  ON agence.coproprietaire
  USING btree
  (reference COLLATE pg_catalog."default");

	type_mouvement character varying not null,

	numero_operation integer not null,
	numero_ligne integer not null,

	liasse_id character varying not null,
	immeuble_id character varying not null,
	lot_id character varying not null,
	base_repart character varying not null,
	nature_id character varying not null,
	coproprietaire_id character varying not null,
	libelle character varying,
	date_reference date,
	debit numeric(11,2),
	credit numeric(11,2),
	solde numeric(11,2),
	statut integer,
	audit_created timestamp without time zone default now(),
	audit_created_by character varying(35),
	audit_updated timestamp without time zone,
	audit_updated_by character varying(35),
	CONSTRAINT operation_pkey PRIMARY KEY (id)
)
WITH (
  OIDS=FALSE
);
ALTER TABLE agence.operation
  OWNER TO postgres;
GRANT ALL ON TABLE agence.operation TO postgres;
GRANT SELECT, UPDATE, INSERT, DELETE ON TABLE agence.operation TO public;

CREATE INDEX operation_liasse_id_idx
  ON agence.operation
  USING btree
  (liasse_id COLLATE pg_catalog."default");

CREATE INDEX operation_type_mouvement_date_operation_type_operation_nume_idx
  ON agence.operation
  USING btree
  (type_mouvement COLLATE pg_catalog."default", date_operation, type_operation COLLATE pg_catalog."default", numero_operation, numero_ligne);
