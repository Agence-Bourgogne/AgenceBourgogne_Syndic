DROP TABLE IF EXISTS agence.exercice_comptable;

CREATE TABLE agence.exercice_comptable
(
  id character varying(32) NOT NULL DEFAULT get_uuid(),
  immeuble_id character varying NOT NULL,
  reference character varying NOT NULL,
  nom character varying NOT NULL,
  date_deb date NOT NULL,
  date_fin date,
  statut integer NOT NULL DEFAULT 0,
  audit_created timestamp without time zone NOT NULL DEFAULT now(),
  audit_created_by character varying,
  audit_updated timestamp without time zone,
  audit_updated_by character varying,
  CONSTRAINT exercice_comptable_pkey PRIMARY KEY (id)
)
WITH (
  OIDS=FALSE
);

ALTER TABLE agence.exercice_comptable
  OWNER TO postgres;
GRANT ALL ON TABLE agence.exercice_comptable TO postgres;
GRANT SELECT, UPDATE, INSERT, DELETE ON TABLE agence.exercice_comptable TO public;

CREATE unique INDEX exercice_comptable_immeuble_id_reference_idx
  ON agence.exercice_comptable
  USING btree
  (immeuble_id COLLATE pg_catalog."default", reference COLLATE pg_catalog."default");


DROP TABLE IF exists agence.budget;

CREATE TABLE agence.budget
(
  id character varying(32) NOT NULL DEFAULT get_uuid(),
  exercice_id character varying NOT NULL,
  reference character varying NOT NULL,
  version integer default 1,
  statut integer,
  audit_created timestamp without time zone DEFAULT now(),
  audit_created_by character varying(35),
  audit_updated timestamp without time zone,
  audit_updated_by character varying(35),
  CONSTRAINT budget_pkey PRIMARY KEY (id)
)
WITH (
  OIDS=FALSE
);
ALTER TABLE agence.budget
  OWNER TO postgres;
GRANT ALL ON TABLE agence.budget TO postgres;
GRANT SELECT, UPDATE, INSERT, DELETE ON TABLE agence.budget TO public;
CREATE UNIQUE INDEX budget_exercice_id_reference_version_idx
  ON agence.budget
  USING btree
  (exercice_id COLLATE pg_catalog."default", reference COLLATE pg_catalog."default", version);
